function fizzbuzz(n) {
  let result = [];
  for (let i = 0; i < n; i++) {
    result.push(i + 1);
    if((result[i] % 3 === 0) &&
      (result[i] % 5 === 0)) {
        result[i] = 'FizzBuzz';
    }
    if(result[i] % 3 === 0) {
      result[i] = 'Fizz';
    }
    if(result[i] % 5 === 0) {
      result[i] = 'Buzz';
    }
  }
  return result;
}
